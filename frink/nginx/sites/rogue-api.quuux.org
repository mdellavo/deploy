
server {
    listen 443 ssl;
    server_name rogue-api.quuux.org;

    ssl_certificate /etc/letsencrypt/live/rogue-api.quuux.org/fullchain.pem;
    ssl_certificate_key /etc/letsencrypt/live/rogue-api.quuux.org/privkey.pem;
    include /etc/letsencrypt/options-ssl-nginx.conf;
    ssl_dhparam /etc/letsencrypt/ssl-dhparams.pem;

    location / {
        proxy_pass http://localhost:6543;
        proxy_set_header Host $host;
        proxy_set_header X-Forwarded-For $remote_addr;
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection "upgrade";
        proxy_buffering off;
        proxy_request_buffering off;
    }
}

server {
    listen 80;
    server_name rogue-api.quuux.org;

    if ($host = rogue-api.quuux.org) {
        return 301 https://$host$request_uri;
    }

    return 404;
}
